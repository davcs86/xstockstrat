## Execution Summary

All work is purely frontend (Next.js + TypeScript). No proto changes, migrations, or config key changes are required — all backend RPCs (`ListBrokerAccounts`, `RegisterBrokerAccount`, `DeregisterBrokerAccount`, `ListPortfolios`) are defined and generated by the `add-ikbr-account-support` feature.

Since the generated TypeScript stubs do not expose broker-account or portfolio RPCs on `main-dev` yet (only `PlaceOrder`, `CancelOrder`, `GetOrder`, `ListOrders`, `StreamOrderUpdates` are in `packages/proto/gen/ts/trading/v1/trading_connect.ts`), all new Connect-RPC calls must use manual service descriptors matching the pattern in `services/xstockstrat-trader/src/lib/connectClients.ts`.

Implementation order: (1) Add broker-account service descriptors to `connectClients.ts`; (2) Add the `AccountContext` React context and provider to `xstockstrat-trader`'s root layout; (3) Add the `/api/accounts` API route; (4) Build the `AccountSelector` and `AccountManagementPanel` components; (5) Update `OrderForm` and `PortfolioSummary`/`OrderBook` to consume selected account; (6) Add the per-account Portfolio tab panel; (7) Wire the `xstockstrat-insights` portfolio selector; (8–9) Add E2E tests for both services.

Steps 1–3 are prerequisites for Steps 4–6. Step 7 is independent of Steps 1–6. Steps 8–9 depend on all prior steps being complete.

## Step Dependencies

- Step 2 requires Step 1: `AccountProvider` must import the `AccountContext` type defined in Step 1's new context file
- Step 3 requires Step 1: the `/api/accounts` route uses the `connectClients.ts` descriptors added in Step 1
- Step 4 requires Steps 2 and 3: `AccountSelector` and `AccountManagementPanel` read from `AccountContext` and call `/api/accounts`
- Step 5 requires Step 2: `OrderForm` and portfolio components read `selectedAccountId` from `AccountContext`
- Step 6 requires Steps 2 and 3: `PortfolioPanel` calls `/api/portfolio/accounts` using selected account
- Step 7 is independent (separate service, no shared context)
- Step 8 requires Steps 1–6: tests cover all trader UI changes
- Step 9 requires Step 7: tests cover insights UI changes

```ts
import { NextRequest, NextResponse } from 'next/server';

const TRADING_BASE_URL =
  process.env.TRADING_HTTP_ENDPOINT ?? 'http://xstockstrat-trading:8051';

async function rpc(method: string, body: object): Promise<Response> {
  return fetch(`${TRADING_BASE_URL}/${method}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/connect+json' },
    body: JSON.stringify(body),
  });
}

// GET /api/accounts — calls ListBrokerAccounts
export async function GET(_req: NextRequest) {
  try {
    const res = await rpc(
      'xstockstrat.trading.v1.TradingService/ListBrokerAccounts',
      {},
    );
    const data = await res.json();
    // Normalize: proto returns accounts[] with snake_case fields (Go uses UseProtoNames: true)
    return NextResponse.json({ accounts: data.accounts ?? [] });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}

// POST /api/accounts — calls RegisterBrokerAccount
export async function POST(req: NextRequest) {
  try {
    const body = await req.json();
    const res = await rpc(
      'xstockstrat.trading.v1.TradingService/RegisterBrokerAccount',
      {
        display_name: body.display_name,
        broker_type: body.broker_type, // 1=ALPACA, 2=IBKR
        is_paper: body.is_paper ?? true,
        credentials_json: body.credentials_json,
      },
    );
    const data = await res.json();
    return NextResponse.json({ account: data.account });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

```ts
import { NextRequest, NextResponse } from 'next/server';

const PORTFOLIO_BASE_URL =
  process.env.PORTFOLIO_HTTP_ENDPOINT ?? 'http://xstockstrat-portfolio:8052';

// GET /api/portfolio/accounts?account_id=XXX — calls ListPortfolios
export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url);
  const accountId = searchParams.get('account_id') ?? '';
  try {
    const res = await fetch(
      `${PORTFOLIO_BASE_URL}/xstockstrat.portfolio.v1.PortfolioService/ListPortfolios`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/connect+json' },
        body: JSON.stringify({ account_id: accountId }),
      },
    );
    const data = await res.json();
    return NextResponse.json({ portfolios: data.portfolios ?? [] });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

import { NextRequest, NextResponse } from 'next/server';

// insights needs to call both trading (for accounts list) and portfolio (for ListPortfolios)
const TRADING_BASE_URL =
  process.env.TRADING_HTTP_ENDPOINT ?? 'http://xstockstrat-trading:8051';
const PORTFOLIO_BASE_URL =
  process.env.PORTFOLIO_HTTP_ENDPOINT ?? 'http://xstockstrat-portfolio:8052';

// GET /api/portfolio?account_id=XXX
// Returns accounts list and per-account portfolios for the insights selector
export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url);
  const accountId = searchParams.get('account_id') ?? '';

  try {
    // Fetch accounts list (for display names and broker types)
    const accountsRes = await fetch(
      `${TRADING_BASE_URL}/xstockstrat.trading.v1.TradingService/ListBrokerAccounts`,
      { method: 'POST', headers: { 'Content-Type': 'application/connect+json' }, body: JSON.stringify({}) },
    );
    const accountsData = await accountsRes.json();

    // Fetch portfolios (filtered by account_id if provided)
    const portfoliosRes = await fetch(
      `${PORTFOLIO_BASE_URL}/xstockstrat.portfolio.v1.PortfolioService/ListPortfolios`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/connect+json' },
        body: JSON.stringify({ account_id: accountId }),
      },
    );
    const portfoliosData = await portfoliosRes.json();

    return NextResponse.json({
      accounts: accountsData.accounts ?? [],
      portfolios: portfoliosData.portfolios ?? [],
    });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
